436  public function updateTranslation($locale, $domain, $key) {
437  list ($type, $oldKey, $oldValue) = explode('||', $key);
438  $oldValue = urldecode($oldValue);
439  $newKey = $_POST['k'];
440  $newValue = urldecode($_POST['v']);
441  $new = URequest::getBoolean('n');
442  $messagesUpdates = new MessagesUpdates($locale, $domain);
443  $messagesUpdates->load();
444  if ($type == 'key') {
445  if ($oldKey === '_new_') {
446  $messagesUpdates->addValue($newKey, $newValue, $oldValue);
447  } elseif ($oldKey !== $newKey) {
448  $messagesUpdates->replaceKey($oldKey, $newKey, $newValue);
449  }
450  } elseif ($type == 'value') {
451  if ($oldValue !== $newValue) {
452  if ($oldKey !== '_new_') {
453  $messagesUpdates->updateValue($newKey, $newValue);
454  } elseif ($newKey != null) {
455  $messagesUpdates->addValue($newKey, $newValue);
456  }
457  }
458  }
459  $messagesUpdates->save();
